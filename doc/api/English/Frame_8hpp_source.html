465template <typename T> bool Frame::is() {
466 switch(this->type()) {
467 case OB_FRAME_IR_LEFT: // Follow
468 case OB_FRAME_IR_RIGHT: // Follow
469 case OB_FRAME_IR:
470 return (typeid(T) == typeid(IRFrame) || typeid(T) == typeid(VideoFrame));
471 case OB_FRAME_DEPTH:
472 return (typeid(T) == typeid(DepthFrame) || typeid(T) == typeid(VideoFrame));
473 case OB_FRAME_COLOR:
474 return (typeid(T) == typeid(ColorFrame) || typeid(T) == typeid(VideoFrame));
475 case OB_FRAME_GYRO:
476 return (typeid(T) == typeid(GyroFrame));
477 case OB_FRAME_ACCEL:
478 return (typeid(T) == typeid(AccelFrame));
479 case OB_FRAME_SET:
480 return (typeid(T) == typeid(FrameSet));
481 case OB_FRAME_POINTS:
482 return (typeid(T) == typeid(PointsFrame));
483 case OB_FRAME_RAW_PHASE:
484 return (typeid(T) == typeid(RawPhaseFrame) || typeid(T) == typeid(VideoFrame));
485 default:
486 std::cout << "ob::Frame::is() did not catch frame type: " << (int)this->type() << std::endl;
487 break;
488 }
489 return false;
490}
