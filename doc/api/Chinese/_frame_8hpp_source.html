442template <typename T> bool Frame::is() {
443 switch(this->type()) {
444 case OB_FRAME_IR_LEFT:
445 case OB_FRAME_IR_RIGHT:
446 case OB_FRAME_IR:
447 return (typeid(T) == typeid(IRFrame) || typeid(T) == typeid(VideoFrame));
448 case OB_FRAME_DEPTH:
449 return (typeid(T) == typeid(DepthFrame) || typeid(T) == typeid(VideoFrame));
450 case OB_FRAME_COLOR:
451 return (typeid(T) == typeid(ColorFrame) || typeid(T) == typeid(VideoFrame));
452 case OB_FRAME_GYRO:
453 return (typeid(T) == typeid(GyroFrame));
454 case OB_FRAME_ACCEL:
455 return (typeid(T) == typeid(AccelFrame));
456 case OB_FRAME_SET:
457 return (typeid(T) == typeid(FrameSet));
458 case OB_FRAME_POINTS:
459 return (typeid(T) == typeid(PointsFrame));
460 default:
461 std::cout << "ob::Frame::is() did not catch frame type: " << (int)this->type() << std::endl;
462 break;
463 }
464 return false;
465}
